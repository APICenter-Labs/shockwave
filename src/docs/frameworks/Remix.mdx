  ## Install

  ```bash
  npm install @apicenterlabs/shockwave
  ```

  ## Example route module

  ```jsx
  import { Button } from '@apicenterlabs/shockwave';

  export default function AccountRoute() {
    return <Button variant="secondary">Save profile</Button>;
  }
  ```
